add_namespace = finland

# Handling the Prisoners #
country_event = {
	id = finland.1
	title = finland.1.t
	desc = finland.1.d
	picture = GFX_report_event_soviet_tanks_snow
	
	is_triggered_only = yes
	
	option = { # Release the Soldiers, Execute Leaders
		name = finland.1.a
		ai_chance = { factor = 50 }
		FIN = {
			remove_ideas = FIN_red_prisoners_of_war
		}
	}
	option = { # Release them all
		name = finland.1.b
		ai_chance = { factor = 50 }
		FIN = {
			remove_ideas = FIN_red_prisoners_of_war
			set_country_flag = released_red_officers
			add_ideas = FIN_red_leaders_pardoned
		}
	}
}

				}

					name = FIN_democratic_party
				}
			}

			set_country_flag = kallio_chosen
			set_politics = {
				ruling_party = democratic
			}
			create_country_leader = {
				name = "Kyösti Kallio"

				picture = "Portrait_Finland_Kyosti_Kallio.dds"

			}

				name = FIN_neutrality_party
			}
			add_popularity = {
				ideology = democratic
				popularity = 0.10
			}
			add_popularity = {
				ideology = communism
				popularity = 0.05
			}
		}
	}
	option = { # Svinhufvud
		name = finland.2.b
		ai_chance = { factor = 50 }

			set_country_flag = svin_chosen

				ruling_party = democratic
			}
			add_popularity = {
				ideology = democratic
				popularity = 0.10
			}
			add_popularity = {
				ideology = communism
				popularity = -0.05
			}
		}
	}
